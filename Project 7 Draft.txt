Enter Continue or Stop: ");		boolean cont = true;		String str = console.nextLine();		if(str.equals("Stop")){		    cont = false;		    System.out.println("Thanks for Visiting A&A Accounting Please allow us to file your 2017 Taxes");		}		return cont;    }    public static void greeting() {	    System.out.println("Welcome to A&A Accounting");	    System.out.println("Glad to help you determine your cost for filing 2012 taxes with us");	 }    public static double billingAmount(double hourlyRate, double consultingTime, double incomeLevel) {		double serviceCharges = 0.0;        if(incomeLevel <= 25000.00){            if(consultingTime > 30.00){                serviceCharges = (hourlyRate * .4) * (consultingTime - 30.00) / 60;            }        }        else{            if(consultingTime > 20.00){                serviceCharges = (hourlyRate * .7) * (consultingTime - 20) / 60;            }        }		return serviceCharges;	}}
